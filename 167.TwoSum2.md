```{python, include = TRUE, message = FALSE}
class Solution(object):
    def twoSum(self, numbers, target):
        """
        :type numbers: List[int]
        :type target: int
        :rtype: List[int]
        """
        d = {}
        for i in range(len(numbers)):
            diff = target - numbers[i]
            if diff in d:
                return [d[diff], i + 1]
            d[numbers[i]] = i + 1
        return list(d)
```
